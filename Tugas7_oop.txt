#Meminta input dari pengguna dalam satuan Kbps
kbps= float(input("masukkan jaringan mbps: "))
mbps = kbps/1000


if mbps > 50:
   kategori ="sangat cepat (A)"
elif mbps > 20:
   kategori ="cepat(B)"
elif mbps > 5:
   kategori ="sedang(C)"
else:
   kategori = "lambat(D) - Memerlukan perbaikan"
   
#Menampilkan hasil keputusan
print("hasil evaluasi :",kategori)
